Props
Styling
Property Type Default Description
currentPage Number 1 The current page number. The default is set to 1 but can be set to any number. This is the page that will be highlighted.
goToButtons String By default, the 'Go to First', 'Previous', 'Next' and 'Go to Last' pagination buttons are displayed as icons. If set to "text", the pagination buttons will be displayed as text.
hideEllipsis Boolean false If set to true, the ellipsis will be hidden.
hideGotoEndButtons Boolean false If set to true, the 'Go to First' and 'Go to Last' buttons will be hidden.
id String The id attribute for the pagination component.
limit Number The number of pages to display. The default is set to 3 but can be set to any number.
pageSizeOptions Array [] The options to be used for the 'Items per page' selection. This is used when showSizeChanger is set to true. The options should be an array of numbers and the default array is '[10, 20, 50, 100, "All"]'.
paginationLayout String The layout to be used for the pagination. The layout should be a string. Options include: for pagination layout 'center', 'end', fill (default is start) OR when used with showSizeChanger 'start', 'center', 'end', 'fill', 'fill-left', 'fill-right'.
paginationVariantColor String The color variant to be used for the pagination. The color variant should be a string. The default is primary with other options: secondary, tertiary, success, danger, warning, and info. Custom colors can be used by passing the color name.
plumage Boolean false If set to true, the pagination will be styled with the Plumage theme.
showDisplayRange Boolean false Shows the display range of the pagination, ex: '1-10 of 22'. If set to true, the display range will be shown.
showSizeChanger Boolean false Shows the 'Items per page' selection. If set to true, the 'Items per page' selection will be shown.
size String The size of the pagination. The size should be a string. Options include: 'sm', 'lg' and no size set by default.
totalPages Number 1 The total number of pages in the table. The pages should be a number.
totalRows Number 0 The total number of rows in the table. The rows should be a number.
useByPagePagination Boolean false If set to true, the user can navigate through the pages using the 'Go to First', 'Previous', 'Next' and 'Go to Last' pagination buttons and an input filed to change the page number.
useMinimizePagination Boolean false If set to true, the pagination will be minimized and only show the 'Go to First', 'Previous', 'Next' and 'Go to Last' buttons.
Using Component in other JavaScript Frameworks

Integrating a Lit Element component with other JavaScript frameworks like Vue, React, and Angular involves creating a wrapper component in each framework to encapsulate and use the Lit Element component. Here are examples for each framework:

Vue.js

Install the necessary dependencies:

          
            npm install lit
            npm install @webcomponents/webcomponentsjs --save

Include the component and polyfills in your project:

            
                <!-- In your index.html -->
              <script src="node_modules/@webcomponents/webcomponentsjs/webcomponents-bundle.js"> </script>
              <script type="module" src="path/to/pagination-component.js"> </script>

Use the component in a Vue component:

            
              <template>
                 <div>
                    <pagination-component
                        paginationLayout="start"
                        :currentPage="currentPage"
                        :totalPages="totalPages"
                        :limit="5"
                        @page-changed="handlePageChange"
                        @page-size-changed="handlePageSizeChange"
                    ></pagination-component>
                  </div>
              </template>
              
                <script>
                    export default {
                        data() {
                        return {
                            currentPage: 1,
                            totalPages: 10
                        };
                        },
                        methods: {
                        handlePageChange(event) {
                            this.currentPage = event.detail.page;
                        },
                        handlePageSizeChange(event) {
                            // Handle page size change
                        }
                        }
                    };
                <script>
React

Install the necessary dependencies:

          
            // bash
            npm install lit
            npm install @webcomponents/webcomponentsjs --save

Include the component and polyfills in your project:

          
            <!-- In your index.html -->
            <script src="node_modules/@webcomponents/webcomponentsjs/webcomponents-bundle.js"></script>
            <script type="module" src="path/to/pagination-component.js"></script>

Use the component in a React component:

            
                import React, { useEffect, useState } from 'react';

                function PaginationExample() {
                  const [currentPage, setCurrentPage] = useState(1);
                  const totalPages = 10;
                
                  useEffect(() => {
                    const handlePageChange = (event) => {
                      setCurrentPage(event.detail.page);
                    };
                
                    const paginationElement = document.querySelector('pagination-component');
                    paginationElement.addEventListener('page-changed', handlePageChange);
                
                    return () => {
                      paginationElement.removeEventListener('page-changed', handlePageChange);
                    };
                  }, []);
                
                  return (
                    <div>
                        <pagination-component
                            paginationLayout="start"
                            currentPage={currentPage}
                            totalPages={totalPages}
                            limit="5"
                        ></pagination-component>
                    </div>
                  );
                }
                
                export default PaginationExample;
Angular

Install the necessary dependencies:

          
            npm install lit
            npm install @webcomponents/webcomponentsjs --save

Include the component and polyfills in your project:

          
            <!-- In your index.html -->
            <script src="node_modules/@webcomponents/webcomponentsjs/webcomponents-bundle.js"></script>
            <script type="module" src="path/to/pagination-component.js"></script>

Use the component in an Angular component:

            
                // In your component.ts file
                import { Component, OnInit } from '@angular/core';
                
                @Component({
                  selector: 'app-pagination-example',
                  template: `
                  <pagination-component
                      paginationLayout="start"
                      [attr.currentPage]="currentPage"
                      [attr.totalPages]="totalPages"
                      [attr.limit]="5"
                      (page-changed)="handlePageChange($event)"
                      (page-size-changed)="handlePageSizeChange($event)"
                   ></pagination-component>
                  `
                })
                export class PaginationExampleComponent implements OnInit {
                  currentPage = 1;
                  totalPages = 10;
                
                  ngOnInit() {}
                
                  handlePageChange(event: CustomEvent) {
                    this.currentPage = event.detail.page;
                  }
                
                  handlePageSizeChange(event: CustomEvent) {
                    // Handle page size change
                  }
                }

Adjust the paths and imports based on your project structure. The provided examples showcase how to create wrapper components for each framework to use your Lit Element 'Pagination' component.

The interoperability is possible because Web Components, once defined and registered as custom elements, act like native HTML elements that these frameworks can understand and interact with.
